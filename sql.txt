CREATE TABLE crm_organization (
	id int NOT NULL AUTO_INCREMENT,
	org_name varchar(255),
	website varchar(255),
        PRIMARY KEY (id)
);

ALTER TABLE organization ADD UNIQUE (
	org_name,
	website
);

	id int NOT NULL AUTO_INCREMENT,
	firstname varchar(255),
	lastname varchar(255),
	email varchar(254),
	phone varchar(24),
        PRIMARY KEY (id)
);
